# ===== Импорты =====
from django.contrib.auth.models import User
from news.models import Author, Category, Post, Comment, PostCategory

# ===== Создание пользователей и авторов =====
user1 = User.objects.create_user('user1', password='pass123')
user2 = User.objects.create_user('user2', password='pass123')
author1 = Author.objects.create(user=user1)
author2 = Author.objects.create(user=user2)

# ===== Создание категорий =====
cat1 = Category.objects.create(name='Спорт')
cat2 = Category.objects.create(name='Политика')
cat3 = Category.objects.create(name='Образование')
cat4 = Category.objects.create(name='Наука')

# ===== Создание постов (2 статьи + 1 новость) =====
post1 = Post.objects.create(author=author1, type='article', title='Первая статья', text='Текст первой статьи...', rating=0)
post2 = Post.objects.create(author=author2, type='article', title='Вторая статья', text='Текст второй статьи...', rating=0)
post3 = Post.objects.create(author=author1, type='news', title='Новость дня', text='Текст новости...', rating=0)

# ===== Присвоение категорий через промежуточную модель PostCategory =====
PostCategory.objects.create(post=post1, category=cat1)
PostCategory.objects.create(post=post1, category=cat2)
PostCategory.objects.create(post=post2, category=cat3)
PostCategory.objects.create(post=post3, category=cat4)

# ===== Создание комментариев (минимум 4) =====
comm1 = Comment.objects.create(post=post1, user=user2, text='Комментарий 1 к первой статье', rating=0)
comm2 = Comment.objects.create(post=post1, user=user1, text='Комментарий 2 к первой статье', rating=0)
comm3 = Comment.objects.create(post=post2, user=user1, text='Комментарий ко второй статье', rating=0)
comm4 = Comment.objects.create(post=post3, user=user2, text='Комментарий к новости', rating=0)

# ===== Применяем like() и dislike() (корректируем рейтинг) =====
post1.like()
post1.dislike()
post2.like()
post3.like()
post3.like()

comm1.like()
comm1.like()
comm2.dislike()
comm3.like()
comm4.like()
comm4.dislike()

# ===== Обновляем рейтинги авторов =====
author1.update_rating()
author2.update_rating()

# ===== Вывод лучшего пользователя (автора) =====
best_author = Author.objects.order_by('-rating').first()
print(f'Лучший пользователь: {best_author.user.username}, рейтинг: {best_author.rating}')

# ===== Вывод лучшей статьи =====
best_post = Post.objects.order_by('-rating').first()
print(f'Дата: {best_post.created_at}')
print(f'Автор: {best_post.author.user.username}')
print(f'Рейтинг статьи: {best_post.rating}')
print(f'Заголовок: {best_post.title}')
print(f'Превью: {best_post.preview()}')

# ===== Вывод всех комментариев к лучшей статье =====
comments = Comment.objects.filter(post=best_post)
for c in comments:
    print(f'{c.created_at} | {c.user.username} | рейтинг: {c.rating} | текст: {c.text}')
